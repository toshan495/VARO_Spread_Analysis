ratio=combined_data['Ratio']

##TEST FOR MEAN REVERSION: HURST EXPONENT
from numpy import cumsum, log, polyfit, sqrt, std, subtract
from numpy.random import randn

def hurst(series):
    lags = range(2, 100) # Create the range of lag values
    tau = [sqrt(std(subtract(series[lag:], series[:-lag]))) for lag in lags] # Calculate the array of the variances of the lagged differences  
    poly = polyfit(log(lags), log(tau), 1) # Use a linear fit to estimate the Hurst Exponent
    he= poly[0]*2.0 # Return the Hurst exponent from the polyfit output
    print("Hurst exponent:",round(he,5))
    print('Time series is mean reverting') if he<0.5 else print('Time series is a Geometric Brownian Motion') if he==0.5 else print('Time series is trending')
    return

print("Mean reversion test for Brent oil")
hurst(brent.to_list())
print("Mean reversion test for WTI oil")
hurst(wti.to_list())
print("Mean reversion test for WTI-Brent spread")
hurst(spread.to_list())

## HALF LIFE CALCULATION BASED ON  Ornstein-Uhlenbeck process
import numpy as np
import statsmodels.api as sm

def halflife(series):
    z_array=series.to_numpy()
    z_lag=np.roll(z_array,1)
    z_lag[0]=0
    z_ret=z_array-z_lag
    z_ret[0]=0
    z_lag2=sm.add_constant(z_lag)
    model=sm.OLS(z_ret,z_lag2)
    res=model.fit()
    h1=-np.log(2)/res.params[1]
    return round(h1,2)
    
print("Half-Life of WTI-Brent spread:",halflife(spread),"days")

#ARIMA MODELING FOR SPREAD FORECASTING

#splitting spread series into testing and training set
df = pd.DataFrame({'Spread': spread.values})
msk = (df.index < len(df)-896)
df_train = df[msk].copy() # training set: 2010 to 2020
df_test = df[~msk].copy() # testing set: 2020 to 2023

#finding parameters for best goodness of fit
import pmdarima as pm
auto_arima = pm.auto_arima(df_train, stepwise=False, seasonal=False)

#fitting model
from statsmodels.tsa.arima.model import ARIMA
model = ARIMA(df_train, order=(0,1,2))
model_fit = model.fit()
print(model_fit.summary())

forecast_test_auto = auto_arima.predict(n_periods=len(df_test))
df['forecast'] = [None]*len(df_train) + list(forecast_test_auto)

#plotting spread series along with predcited values
